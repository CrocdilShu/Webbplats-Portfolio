Projekt 01 MicroPython WS2812 PIO

NeoPixel Clock

En analog klocka med 24 WS2812 NeoPixel-LEDs. Timmar visas i röd, minuter i gul och sekunder i grön. PIO hanterar det tidskritiska WS2812-protokollet.

Komponenter

  • Raspberry Pi Pico
  • 24× WS2812 NeoPixel LEDs
  • RTC (Real-Time Clock)

Färgkodning

  • Röd — Timmar
  • Gul — Minuter
  • Grön — Sekunder
neopixel.py 
# Example using PIO to drive a set of WS2812 LEDs.

import array, time
from machine import Pin, RTC
import rp2
rtc = RTC()
NUM_LEDS = 24
PIN_NUM = 6
brightness = 0.02

@rp2.asm_pio(sideset_init=rp2.PIO.OUT_LOW,
             out_shiftdir=rp2.PIO.SHIFT_LEFT,
             autopull=True, pull_thresh=24)
def ws2812():
    T1 = 2
    T2 = 5
    T3 = 3
    wrap_target()
    label("bitloop")
    out(x, 1)               .side(0)    [T3 - 1]
    jmp(not_x, "do_zero")   .side(1)    [T1 - 1]
    jmp("bitloop")          .side(1)    [T2 - 1]
    label("do_zero")
    nop()                   .side(0)    [T2 - 1]
    wrap()

sm = rp2.StateMachine(0, ws2812, freq=8_000_000,
                       sideset_base=Pin(PIN_NUM))
sm.active(1)
ar = array.array("I", [0 for _ in range(NUM_LEDS)])

def pixels_show():
    dimmer_ar = array.array("I", [0 for _ in range(NUM_LEDS)])
    for i, c in enumerate(ar):
        r = int(((c >> 8) & 0xFF) * brightness)
        g = int(((c >> 16) & 0xFF) * brightness)
        b = int((c & 0xFF) * brightness)
        dimmer_ar[i] = (g<<16) + (r<<8) + b
    sm.put(dimmer_ar, 8)
    time.sleep_ms(10)

def pixels_set(i, color):
    ar[i] = (color[1]<<16) + (color[0]<<8) + color[2]

def pixels_fill(color):
    for i in range(len(ar)):
        pixels_set(i, color)

BLACK = (0, 0, 0)
RED = (255, 0, 0)
YELLOW = (255, 150, 0)
GREEN = (0, 255, 0)

while True:
    t = rtc.datetime()
    h, m, s = t[4], t[5], t[6]
    pixels_fill(BLACK)
    sec_led = int(s / 5) * 2
    min_led = int(m / 5) * 2
    hour_led = (h % 12) * 2
    pixels_set(sec_led, GREEN)
    pixels_set(min_led, YELLOW)
    pixels_set(hour_led, RED)
    pixels_show()
    time.sleep(0.1)
Projekt 02 MicroPython 7-segment OLED

Home/Away Scoreboard

En poängtavla med MAX7219 7-segment display och SSD1306 OLED. Sex knappar styr poäng, reset och goal-animation med buzzer-ljudeffekt.

Komponenter

  • Raspberry Pi Pico
  • MAX7219 7-segment
  • SSD1306 OLED 128×64
  • Piezo-buzzer
  • 6× knappar

Protokoll

  • SPI — Display
  • I2C — OLED
  • PWM — Buzzer

Kopplingsschema i Fritzing

HomeAway-2.py 
from machine import Pin, SPI, PWM, I2C
import max7219_8digit
import ssd1306
import time

spi = SPI(0, baudrate=10000000, polarity=1, phase=0,
          sck=Pin(2), mosi=Pin(3))
ss = Pin(5, Pin.OUT)
display = max7219_8digit.Display(spi, ss)

i2c = I2C(0, scl=Pin(1), sda=Pin(0), freq=400000)
oled = ssd1306.SSD1306_I2C(128, 64, i2c)
oled.fill(0)
oled.show()

home_plus  = Pin(6,  Pin.IN, Pin.PULL_UP)
home_minus = Pin(7,  Pin.IN, Pin.PULL_UP)
away_plus  = Pin(8,  Pin.IN, Pin.PULL_UP)
away_minus = Pin(9,  Pin.IN, Pin.PULL_UP)
reset_btn  = Pin(10, Pin.IN, Pin.PULL_UP)
goal_btn   = Pin(11, Pin.IN, Pin.PULL_UP)

buzzer = PWM(Pin(16))
buzzer.duty_u16(0)

home_score = 0
away_score = 0
display.write_to_buffer("H-00A-00")
display.display()

while True:
    if home_plus.value() == 0:
        if home_score < 99:
            home_score += 1
        time.sleep(0.2)

    if home_minus.value() == 0:
        if home_score > 0:
            home_score -= 1
        time.sleep(0.2)

    if away_plus.value() == 0:
        if away_score < 99:
            away_score += 1
        time.sleep(0.2)

    if away_minus.value() == 0:
        if away_score > 0:
            away_score -= 1
        time.sleep(0.2)

    if reset_btn.value() == 0:
        home_score = 0
        away_score = 0
        time.sleep(0.2)

    if goal_btn.value() == 0:
        buzzer.duty_u16(30000)
        for f in [600, 900, 1200]:
            buzzer.freq(f)
            time.sleep(0.2)
        buzzer.duty_u16(0)
        oled.fill(1)
        oled.text("GOAL!", 44, 28, 0)
        oled.show()
        time.sleep(2)
        oled.fill(0)
        oled.show()

    display_text = "H-{:02d}A-{:02d}".format(
        home_score, away_score)
    display.write_to_buffer(display_text)
    display.display()
    time.sleep(0.01)
